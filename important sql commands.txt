create personaldetails table command in postgresql ===>

create table personaldetails(id serial NOT NULL PRIMARY KEY, firstname varchar (100) NOT NULL,lastname varchar (100) NOT NULL, mobilenumber varchar(13), email varchar (100) NOT NULL);


create user table command ===>
create table useraddress(userid integer,address varchar(150),city varchar(40),country varchar(20),state varchar(30), zipcode integer);


join command ===>
SELECT * FROM personaldetails left JOIN useraddress on personaldetails.id = useraddress.userid;



alter id sequence ===>
alter sequence personaldetails_id_seq restart with 1;
